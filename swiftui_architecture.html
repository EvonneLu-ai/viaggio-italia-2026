Swift Struct 定義
// Models/TripItinerary.swift
import Foundation
import CoreLocation

// ── 行程主體 ──────────────────────────────────────
struct TripItinerary: Identifiable, Codable {
  var id: UUID = UUID()
  var title: String            // "2026 南義之旅"
  var startDate: Date
  var endDate: Date
  var days: [DaySchedule]
  var travelers: Int          // 2
  var baseCurrency: String   // "EUR"
  var homeCurrency: String   // "TWD"
}

// ── 每日行程 ──────────────────────────────────────
struct DaySchedule: Identifiable, Codable {
  var id: UUID = UUID()
  var date: Date
  var city: String            // "Roma"
  var emoji: String           // "🏛️"
  var attractions: [Attraction]
  var accommodation: Accommodation?
  var transport: [TransportSegment]
  var notes: String
}

// ── 景點 ──────────────────────────────────────────
struct Attraction: Identifiable, Codable {
  var id: UUID = UUID()
  var name: String
  var nameLocal: String      // 義大利文名稱
  var emoji: String
  var coordinate: GeoCoordinate
  var address: String
  var visitTime: Date?
  var stayDuration: TimeInterval // 建議停留時間（秒）
  var openingHours: String
  var ticketPrice: Double?     // EUR，nil = 免費
  var bookingCode: String?
  var bookingURL: URL?
  var dressCode: DressCode?
  var visitNotes: [VisitNote]
  var nearbyPOI: [POI]
  var checkInLeadTime: Int?   // 提前幾分鐘到（如 30）
}
// Models/SupportingTypes.swift
// ── 周邊標的 ──────────────────────────────────────
struct POI: Identifiable, Codable {
  var id: UUID = UUID()
  var type: POIType           // .restaurant, .cafe, .atm...
  var name: String
  var distance: Double        // 公尺
  var rating: Double?         // 0–5
  var coordinate: GeoCoordinate
  var googleMapsURL: URL?
}
enum POIType: String, Codable, CaseIterable {
  case restaurant, cafe, atm, pharmacy, supermarket, toilet, souvenir, hotel
  var icon: String {
    switch self {
    case .restaurant: return "🍝"
    case .cafe: return "☕"
    case .atm: return "🏧"
    case .pharmacy: return "💊"
    case .toilet: return "🚻"
    default: return "📍"
    }
  }
}

// ── 花費 ──────────────────────────────────────────
struct Expense: Identifiable, Codable {
  var id: UUID = UUID()
  var title: String
  var category: ExpenseCategory
  var amount: Double          // 以 EUR 為主
  var currency: String        // "EUR" / "TWD"
  var amountTWD: Double       // 換算台幣
  var date: Date
  var city: String
  var note: String
}
enum ExpenseCategory: String, Codable, CaseIterable {
  case flight, train, accommodation, food, ticket, shopping, transport, car, misc
}

// ── 語言速查 ──────────────────────────────────────
struct Phrasebook: Identifiable, Codable {
  var id: UUID = UUID()
  var situation: PhraseSituation
  var chinese: String
  var english: String
  var italian: String
  var pronunciation: String   // 音標
  var isFavorite: Bool = false
}
enum PhraseSituation: String, Codable, CaseIterable {
  case accommodation, dining, transport, shopping, emergency, greeting, museum
}

// ── 近期活動 ──────────────────────────────────────
struct LocalEvent: Identifiable, Codable {
  var id: UUID = UUID()
  var title: String
  var date: Date
  var city: String
  var category: String         // "festival", "concert"...
  var ticketURL: URL?
  var isFree: Bool
}

框架SwiftUI、SwiftData（讀取行程）、CoreLocation（目前位置）
ViewModelItineraryViewModel
特色功能已過時間點變暗（opacity 0.4）、目前景點高亮、點擊進入 DayDetailView
📷CameraTranslationView — AR 相機翻譯
UI 元素AVCaptureVideoPreviewLayer 全螢幕相機、翻譯結果 Overlay（CATextLayer）、凍結畫面按鈕、翻譯歷史 Sheet
框架Vision（VNRecognizeTextRequest）、AVFoundation（AVCaptureSession）、Translation framework
核心邏輯每 0.5 秒截取一幀 → Vision OCR → 取得文字邊框 → Translation API → 疊加翻譯文字在原位
特色功能支援「凍結模式」（拍下來慢慢看）、收藏片語、離線模式（系統翻譯語言包需預先下載）
💶ExpenseView — 記帳主頁
UI 元素月份總覽 Header、Swift Charts 圓餅圖（類別佔比）、每日長條圖、最近記錄 List、新增花費 FAB 按鈕
框架Swift Charts、SwiftData（Expense 查詢）、URLSession（匯率 API）
特色功能EUR / TWD 切換顯示、預算超支警示（紅色 Banner）、匯率顯示（1 EUR = 35 NTD）
圖表SectorMark（圓餅）+ BarMark（長條）+ LineMark（累計折線）
🗺️MapView — 即時地圖
UI 元素Map（MapKit）全螢幕、行程景點 Annotation（自訂圖釘）、路線 MKPolyline、底部可拖曳 Sheet（周邊 POI 清單）
框架MapKit（iOS 17 新 API）、CoreLocation、MKLocalSearch（周邊搜尋）
特色功能城市離線地圖下載（.mapTileOverlay）、大眾運輸/步行/開車三種路線模式切換
